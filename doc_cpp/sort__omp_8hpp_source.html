69  const auto median3 = [&](const auto& a, const auto& b, const auto& c, const Compare& comp) {
70  return comp(a, b) ? (comp(b, c) ? b : (comp(a, c) ? c : a)) : (comp(a, c) ? a : (comp(b, c) ? c : b));
71  };
